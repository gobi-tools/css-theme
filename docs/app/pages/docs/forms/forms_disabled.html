A form can have all or part of its inputs set as disabled to prevent any user interaction.

Disabled form
<form>
  <fieldset>
    <legend>
      Disabled form
    </legend>

    <label for="email">
      Email
    </label>
    <input 
      type="email" 
      id="email" 
      placeholder="N/A" 
      disabled/>
    
    <label for="address">
      Address
    </label>
    <input 
      type="text" 
      id="address" 
      placeholder="Address" 
      disabled/>
    
    <label form="delivery">
      Delivery
    </label>
    <select 
      id="delivery" 
      disabled>
      <option 
        value="fast" 
        selected>
        Fast
      </option>
      <option 
        value="standard">
        Standard
      </option>
    </select>
    
    <input 
      type="submit" 
      value="Submit" 
      disabled/>
  </fieldset>
</form>
